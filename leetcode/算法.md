## dijkstra 算法
贪心算法：每一步都获取当前最优解
~~~go
    graph := make([][]int, n)
    // ...
    dist := make([]int, n)
	for s := 0; s < n; s++ {
		dist[s] = inf
	}
	dist[k-1] = 0
	done := make([]bool, n)
	for t := 0; t < n; t++ {
		now := -1
		for i, v := range done {
			if !v && (now == -1 || dist[i] < dist[now]) {
				now = i
			}
		}
		done[now] = true
		for e, l := range graph[now] {
			dist[e] = min(dist[e], dist[now] + l)
		}
	}
~~~
